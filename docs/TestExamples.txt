select AHI.ACT_NAME_,AHI.`START_TIME_`,AHI.`ACT_TYPE_`,pi.`BUSINESS_KEY_` 
from ACT_HI_ACTINST as AHI where AHI.`PROC_INST_ID_` in 
	(select AHP.`PROC_INST_ID_` from ACT_HI_PROCINST as AHP where AHP.`BUSINESS_KEY_`=0001332116)
and 
AHI.`ACT_TYPE_` in ('intermediateNoneThrowEvent','startEvent','endEvent')
